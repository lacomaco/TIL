class Lock{
    constructor(){
        this._lock = false;
        this._waiting = [];
    }
    lock (){

        const unlock = ()=>{
            let nextResolve
            if(this._waiting>length>0){
                nextResolve = this._waiting.pop(0);
                nextResolve(unlock);
            }else{
                this._locked=false;
            }
        }

        //락 되어있을경우
        if(this._locked){
            return new Promise(resolve=>{
                this._waiting.push(resolve);
            });
        }else{
            //락이 안되어잇을경우 resolve를 풀어내야한다.
            this._locked = true;
            return new Promise(resolve=>{
               resolve(unlock);
            });
        }

    }
}
